```python
"""Demonstrate: Guardrail intercepts dangerous commands."""
from harness.guardrail import Guardrail
from harness.config import GuardrailConfig
from harness.models import ToolCall

guard = Guardrail(GuardrailConfig())

# Should be blocked
result = guard.check_tool_call(ToolCall(id="1", name="run_shell", arguments={"command": "rm -rf /"}))
assert not result.allowed
print(f"[BLOCKED] {result.reason}")

# Should be allowed
result = guard.check_tool_call(ToolCall(id="2", name="run_shell", arguments={"command": "pytest tests/"}))
assert result.allowed
print("[ALLOWED] Safe command passed")
```

### 演示 3：自适应修复完整周期（`demo_autonomous_repair.py`）

```python
"""Demonstrate: Full TDD cycle with autonomous repair using MockProvider."""
from harness.config import Config
from harness.context import ContextManager
from harness.guardrail import Guardrail
from harness.loop import HarnessLoop
from harness.memory import MemoryStore
from harness.models import LLMResponse, ToolCall, ToolDef
from harness.stop_condition import AutonomousStopDecision
from providers.factory import ProviderFactory
from providers.mock import MockProvider
from feedback.engine import FeedbackEngine
from tools.dispatcher import ToolDispatcher

# Register tools
dispatcher = ToolDispatcher()
from tools.read_file import ReadFile
from tools.write_file import WriteFile
from tools.run_shell import RunShell
dispatcher.register(ReadFile())
dispatcher.register(WriteFile())
dispatcher.register(RunShell())

# Create MockProvider with a cycle: write code → test fails → finish
provider = MockProvider(preset_responses={
    "Write a Fibonacci function": MockProvider.with_tool_call("write_file", {"path": "fib.py", "content": "def fib(n): ..."}),
    "": MockProvider.with_tool_call("run_shell", {"command": "pytest test_fib.py"}),
})

config = Config()
guardrail = Guardrail(config.guardrail)
stop = AutonomousStopDecision(guardrail, config.loop)
context = ContextManager()
memory = MemoryStore()
feedback = FeedbackEngine()

loop = HarnessLoop(
    provider=provider,
    dispatcher=dispatcher,
    guardrail=guardrail,
    context_manager=context,
    stop_decision=stop,
    feedback_engine=feedback,
    memory_store=memory,
    config=config,
)

tool_defs = [
    ToolDef(name="read_file", description="Read a file", parameters={...}),
    ToolDef(name="write_file", description="Write a file", parameters={...}),
    ToolDef(name="run_shell", description="Run a shell command", parameters={...}),
]

result = loop.run("task", tool_defs)
print(f"Result: {'SUCCESS' if result.success else 'FAILURE'} ({result.iterations} iterations)")
```

**注意：** `demo_autonomous_repair.py` 中的 MockProvider 需要根据实际 HarnessLoop 的测试命令识别逻辑调整 preset 触发方式。关键是展示完整的"写代码→测试→修复→通过"流程。
